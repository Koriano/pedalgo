Algo nombreMystere

Var
	x:ENTIER
	CONST num_mystere:ENTIER<-69
FinVar

Meth
	Fonction ENTIER aleat()
		retourne 42
	FinFonction
FinMeth

Debut
	num_mystere<-aleat()

	TantQue NON (x==num_mystere)
		x<-lireEntier()

		Si ((x<0) OU (x>100))
			afficherTexte("Veuillez saisir un entier entre 0 et 100")
		Sinon 
            Si (x>num_mystere)
                afficherTexte("Le chiffre est plus petit !")
            Sinon 
                Si (x<num_mystere)
                    afficherTexte("Le chiffre est plus grand !")
                Sinon 
                    afficherTexte("Bravo, vous avez gagnÃ© !")
                    FinSi
                FinSi
            FinSi
        FinSi
	FinTantQue
Fin
